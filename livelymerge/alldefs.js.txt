  };
  world.onPointerUp = function (p, e) {
    let clickDuration = e.timeStamp - world.clickTime;
    /*console.log('Click duration = ' + clickDuration);
    if (clickDuration < 150) {
      // Special "quickClick" event for, eg, halo access
      clearTimeout(world.clickPending);
      world.onQuickClick(p, e);
      return;
    } */
    // console.log("pu", x, y);
    this.topLevelMorph.onPointerUp(p, e);
  };
  world.onPointerMove = function (p, e) {
    // console.log("pm", x, y, e.button, e.buttons, e.offsetX, e.offsetY);
    this.topLevelMorph.onPointerMove(p, e);
  };
  world.onQuickClick = function (p, e) {
    // console.log("pm", x, y, e.button, e.buttons, e.offsetX, e.offsetY);
    this.topLevelMorph.onQuickClick(p, e);
  };

  // keyboard events
  world.onKeyPress = function (e) {
    // console.log('TODO(dan): handle key onKeyPress event', e);
    // this.topLevelMorph.onKeyPress(e);
  };
  world.onKeyDown = function (e) {
    // console.log('TODO(dan): handle key press event', e);
    // if (e.key == 'Meta') return; // ignore; check evt.metaKey on simple chars
    this.topLevelMorph.onKeyDown(e);
  };
  world.onKeyUp = function (e) {
    // console.log('TODO(dan): handle key up event', e);
    //this.topLevelMorph.onKeyUp(e);
  };

  let canvasEvents = [];
  const canvas = document.querySelector('canvas');
  canvas.addEventListener('pointerdown', (e) => canvasEvents.push(e));
  canvas.addEventListener('pointerup', (e) => canvasEvents.push(e));
  canvas.addEventListener('pointermove', (e) => canvasEvents.push(e));
  canvas.tabIndex = 1;
  canvas.addEventListener('keydown', (e) => canvasEvents.push(e));
  canvas.addEventListener('keypress', (e) => canvasEvents.push(e));
  canvas.addEventListener('keyup', (e) => canvasEvents.push(e));

  window.renderContinually = true;

  window.renderOnce = () => {
    impl.change(() => {
      processEvents();
      if (world.render) {
        world.render();
      }
    });
  };

  function onFrame() {
    if (renderContinually) {
      try {
        renderOnce();
      } catch (e) {
        console.error(e);
        debugger;
      }
    }
    requestAnimationFrame(onFrame);
  }

  requestAnimationFrame(onFrame);

  function processEvents() {
    for (const e of canvasEvents) {
      e.actorId = window.actorId;
      // console.log('processing', e);
      // TODO: rethink this wrt world.pt
      switch (e.type) {
        case 'pointerdown': {
          const pt = world.pt(e.offsetX, e.offsetY);
          console.log('pointer down point is', pt._id);
          world.onPointerDown(pt, e);
          break;
        }
        case 'pointerup': {
          const pt = world.pt(e.offsetX, e.offsetY);
          // console.log('pointer up point is', pt._id);
          world.onPointerUp(pt, e);
          break;
        }
        case 'pointermove': {
          const pt = world.pt(e.offsetX, e.offsetY);
          // console.log('pointer move point is', pt._id);
          world.onPointerMove(pt, e);
          break;
        }
        case 'keydown':
          world.onKeyDown(e);
          break;
        case 'keypress':
          world.onKeyPress(e);
          break;
        case 'keyup':
          world.onKeyUp(e);
          break;
        default:
          console.error('unsupported even type', e.type);
      }
    }
    canvasEvents = [];
  }
};

world.render = function () {
  const canvas = document.querySelector('canvas');
  if (!canvas || !world.topLevelMorph || !world.topLevelMorph.render) {
    return;
  }

  const ctx = canvas.getContext('2d');
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  world.topLevelMorph.render(ctx);
};
world.canvas = () => document.querySelector('canvas');
world.getBounds = function () {
  const canvas = world.canvas();
  if (!canvas) {
    return;
  }
  return world.rect(0, 0, canvas.width, canvas.height);
};

// Graphics

world.Point = world.newClass('Point');
world.Point.proto.initialize = function (x, y) {
  this.x = x;
  this.y = y;
};
world.Point.proto.copy = function () {
  return world.pt(this.x, this.y);
};
world.Point.proto.addPt = function (p) {
  return world.pt(this.x + p.x, this.y + p.y);
};
world.Point.proto.subPt = function (p) {
  return world.pt(this.x - p.x, this.y - p.y);
};
world.Point.proto.lePt = function (p) {
  return this.x <= p.x && this.y <= p.y;
};
world.Point.proto.moveBy = function (p) {
  // ** why are we side-effecting here, but copying in other methods?
  this.x += p.x;
  this.y += p.y;
};
world.Point.proto.scaleBy = function (s) {
  const sp = typeof s === 'number' ? world.pt(s, s) : s;
  return world.pt(this.x * sp.x, this.y * sp.y);
};
world.Point.proto.asString = function () {
  return `pt(${this.x.toFixed(2)}, ${this.y.toFixed(2)})`;
};
world.Point.proto.toString = function () {
  return this.asString();
};
world.Point.proto.minPt = function (p) {
  return world.Point.new(Math.min(this.x, p.x), Math.min(this.y, p.y));
};
world.Point.proto.maxPt = function (p) {
  return world.Point.new(Math.max(this.x, p.x), Math.max(this.y, p.y));
};
world.Point.proto.extent = function (ext) {
  // Make a rectangle
  return world.Rectangle.new(this, ext);
};
world.Point.proto.dist = function (p) {
  var dx = p.x - this.x;
  var dy = p.y - this.y;
  return Math.sqrt(dx * dx + dy * dy);
};
world.Point.proto.polarAngle = function () {
  return Math.atan2(this.x, this.y);
};
world.Point.proto.rotatedBy = function (radians, pivotPt) {
  let v = this.subPt(pivotPt);
  let polarDist = Math.sqrt(v.x * v.x + v.y * v.y);
  let rot = world.ptPolar(polarDist, v.polarAngle() + radians);
  return rot.addPt(pivotPt);
};
world.Point.proto.render = function (ctx) {
  // ctx.beginPath();
  ctx.fillStyle = world.Color.black.fillStyle;
  ctx.arc(this.x, this.y, 8, 0, 2 * Math.PI);
  ctx.fill();
};

world.Rectangle = world.newClass('Rectangle');
world.Rectangle.proto.initialize = function (p, ext) {
  this.topLeft = p;
  this.extent = ext;
};
world.Rectangle.proto.copy = function () {
  return this.topLeft.copy().extent(this.extent.copy());
};
world.Rectangle.proto.bottomLeft = function () {
  return this.topLeft.addPt(world.pt(0, this.height()));
};
world.Rectangle.proto.topRight = function () {
  return this.topLeft.addPt(world.pt(this.width(), 0));
};
world.Rectangle.proto.bottomRight = function () {
  return this.topLeft.addPt(this.extent);
};
world.Rectangle.proto.moveBy = function (p) {
  //NOTE: this does not return a copy, but *changes this rect*
  this.topLeft = this.topLeft.addPt(p);
};
world.Rectangle.proto.movedBy = function (p) {
  //NOTE: this *does* return a copy, and doesnt change this rect
  return this.topLeft.addPt(p).extent(this.extent.copy());
};
world.Rectangle.proto.asString = function () {
  return `${this.topLeft.asString()}.extent(${this.extent.asString()})`;
};
world.Rectangle.proto.toString = function () {
  return this.asString();
};
world.Rectangle.proto.renderOn = function (ctx) {
  //console.log('rendering rectangle', this._id);
  // this.render(ctx, 'white', 1, 'black')
};
world.Rectangle.proto.render = function (ctx, fillColor, borderWidth, borderColor) {
  if (fillColor) {
    ctx.fillStyle = fillColor.fillStyle;
    ctx.fillRect(this.topLeft.x, this.topLeft.y, this.extent.x, this.extent.y);
  }
  if (borderWidth) {
    borderColor = borderColor ? borderColor : world.Color.black;
    ctx.fillStyle = borderColor.fillStyle;
    ctx.lineWidth = borderWidth ? borderWidth : 2;
    ctx.strokeRect(this.topLeft.x, this.topLeft.y, this.extent.x, this.extent.y);
  }
};
world.Rectangle.proto.center = function (p) {
  return this.topLeft.addPt(world.pt(this.extent.x / 2, this.extent.y / 2));
};
world.Rectangle.proto.insetBy = function (numOrPt) {
  const inset = typeof numOrPt === 'number' ? world.pt(numOrPt, numOrPt) : numOrPt;
  return this.topLeft.addPt(inset).extent(this.extent.subPt(inset).subPt(inset));
};
world.Rectangle.proto.width = function () {
  return this.extent.x;
};
world.Rectangle.proto.height = function () {
  return this.extent.y;
};
world.Rectangle.proto.setBounds = function (rect) {
  this.topLeft = rect.topLeft;
  this.extent = rect.extent;
};
world.Rectangle.proto.getBounds = function () {
  return world.Rectangle.new(this.topLeft, this.extent);
};
world.Rectangle.proto.includesPt = function (p) {
  return this.topLeft.lePt(p) && p.lePt(this.bottomRight());
};
world.Rectangle.proto.scaleRect = function (spec) {
  // spec is a rectangle giving a relative placement of, eg a browser pane
  let topLeft = this.topLeft.addPt(spec.topLeft.scaleBy(this.extent));
  let extent = this.extent.scaleBy(spec.extent);
  return topLeft.extent(extent);
};
world.Rectangle.proto.onPointerDown = function (p, e) {
  if (this.actorId != null && this.actorId != window.actorId) return false;
  if (this.includesPt(p)) {
    this.hitPoint = p;
    this.actorId = e.actorId;
    return true;
  }
  return false;
};
world.Rectangle.proto.onPointerMove = function (p, e) {
  if (this.actorId != null && this.actorId != window.actorId) return;
  if (this.hitPoint) {
    this.moveBy(p.subPt(this.hitPoint));
    this.hitPoint = p;
  }
};
world.Rectangle.proto.onPointerUp = function (p) {
  if (this.actorId != null && this.actorId != window.actorId) return;
  this.actorId = null;
  delete this.hitPoint;
};
world.Rectangle.unionPts = function (points) {
  // points is a List
  let tl = points.first();
  let br = points.first();
  points.forEach((each) => {
    tl = tl.minPt(each);
    br = br.maxPt(each);
  });
  return world.Rectangle.new(tl, br.subPt(tl));
};

world.PolyLine = world.Shape.subclass('PolyLine');
world.PolyLine.proto.initialize = function (verts, width, color) {
  this.vertices = verts;
  let bounds = world.Rectangle.unionPts(verts);
  console.log('Polyline bounds = ', bounds.asString());
  world.Shape.proto.initialize.call(this, 'PolyLine', bounds, null, width, color);
};
world.PolyLine.proto.copy = function () {
  let copy = world.PolyLine.new(this.vertices.copy(), this.borderWidth, this.borderColor);
  copy.setStyles(this.fillColor, this.borderWidth, this.borderColor);
  return copy;
};
world.PolyLine.proto.asString = function () {
  return `PolyLine at ${this.topLeft.asString()} with size ${this.extent}`;
};
world.PolyLine.proto.render = function (ctx) {
  ctx.beginPath();
  let first = true;
  this.vertices.forEach((vert) => {
    if (first) {
      ctx.moveTo(vert.x, vert.y);
      first = false;
    } else {
      ctx.lineTo(vert.x, vert.y);
    }
  });
  if (this.fillColor !== null) {
    ctx.fillStyle = this.fillColor.fillStyle;
    ctx.fill();
  }
  if (this.borderColor !== null) {
    ctx.lineWidth = this.borderWidth;
    ctx.strokeStyle = this.borderColor.fillStyle;
    ctx.stroke();
  }
  ctx.closePath();
};
world.PolyLine.proto.renderOn = function (ctx) {
  // console.log('rendering Shape', this.shapeType);
  this.render(ctx);
};
world.PolyLine.proto.moveBy = function (d) {
  world.Shape.proto.moveBy.call(this, d); // moves the bounds
  this.vertices = this.vertices.map((vert) => vert.addPt(d));
};
world.PolyLine.proto.onPointerMove = function (p) {
  if (this.actorId != null && this.actorId != window.actorId) {
    return;
  }
  if (this.hitPoint) {
    const d = p.subPt(this.hitPoint);
    // *** isn't this just this.moveBy??
    this.moveBy(d); // moves the bounds
    this.vertices = this.vertices.map((vert) => vert.addPt(d));
    this.hitPoint = p;
  }
};
world.PolyLine.proto.rotateBy = function (radians, pivotPt) {
  let pivot = pivotPt ? pivotPt : this.center();
  this.vertices = this.vertices.map((vert) => {
    return vert.rotatedBy(radians, pivot);
  });
};

world.Pen.proto.makeHandShape = function (location, color) {
  this.setPenColor(color);
  this.location = location;
  this.vertices = world.List.new(this.location);
  for (let i = 1; i <= 3; i++) {
    this.go(20);
    this.turn(120);
  }
  let handShape = this.polyLine();
  handShape.setColor(color);
  handShape.setBorderWidth(1);
  handShape.setBorderColor(world.Color.black);
  return handShape;
};
world.Pen.proto.spiral = function (n, d, a) {
  this.setPenColor(world.Color.red);
  for (let i = 1; i <= n; i++) {
    this.go(d * i);
    this.turn(a);
  }
  return this.polyLine();
};
world.Pen.proto.star = function (nVerts, radius, colr) {
  // Really belongs in PolyLine
  let vertices = world.List.new();
  for (let i = 0; i <= nVerts; i++) {
    var angle = ((2 * Math.PI) / nVerts) * i;
    var p = world.ptPolar(radius, angle);
    if (i % 2 == 0) p = p.scaleBy(0.39);
    vertices.push(p.addPt(this.location));
  }
  return world.PolyLine.new(vertices, 2, world.Color.black);
};

world.TextBox.proto.compose = function () {
  // This version will break not only at newLines, but also
  // at any alphanumeric preceded by a non-alphanumeric.
  // That covers normal 'word-breaks', but also, eg, long JS epressions.
  // I'm sure regex magic could help, but this is readable
  this.lines = world.List.new();
  this.lineStarts = world.List.new();
  let ctx = this.getTextContext(this.font);
  let str = this.string;
  let lineStart = 0;
  let newline = 13;
  let lineFeed = 10;
  let lineTopLeft = this.topLeft.addPt(world.pt(2, 2));
  let lineNo = 0;
  let inAlpha = false; // means we are in middle of an identifier
  let alphaBreak = 0; // most recent start of an identifier
  for (let idx = 0; idx < str.length; idx++) {
    let c = str[idx];
    let isAlpha = /^[a-zA-Z0-9]*$/.test(c);
    // console.log(`str[idx] = ${str[idx]}; charCode = ${c}`)
    if (c == '\n' || c == '\r' || idx == str.length - 1) {
      // Break after newLine (also end of string)
      let thisLine = world.TextLine.new(
        lineTopLeft,
        world.pt(this.extent.x, this.lineHeight),
        str.slice(lineStart, idx + 1),
        this.font,
        world.Color.blue
      );
      this.lines.push(thisLine);
      this.lineStarts.push(lineStart);
      lineNo++;
      lineTopLeft = lineTopLeft.addPt(world.pt(0, this.lineHeight));
      lineStart = idx + 1;
    } else {
      if (!inAlpha && isAlpha) {
        // This is a 'word break'.  Check if we're out of bounds...
        let maybeLine = str.slice(lineStart, idx + 1); // leaves off the newline
        let metrics = ctx.measureText(maybeLine);
        if (metrics.width >= this.extent.x) {
          // We're out of bounds; Need to issue a line up to previous break
          let thisLine = world.TextLine.new(
            lineTopLeft,
            world.pt(this.extent.x, this.lineHeight),
            str.slice(lineStart, alphaBreak),
            this.font,
            world.Color.blue
          );
          this.lines.push(thisLine);
          this.lineStarts.push(lineStart);
          lineNo++;
          lineTopLeft = lineTopLeft.addPt(world.pt(0, this.lineHeight));
          lineStart = alphaBreak;
        }
        alphaBreak = idx; // Note this break and carry on
        inAlpha = true;
      }
      inAlpha = isAlpha; // Track when we are 'in' an identifier
    }
  }
  return lineTopLeft.y + 2;
};
world.TextBox.proto.getTextContext = function (font) {
  const canvas = document.querySelector('canvas');
  if (!canvas) return;
  const ctx = canvas.getContext('2d');

};
world.TextBox.proto.xValuesForLine = function (lineNo) {
  // NOTE:  this should receive a line (this.lines.at(lineNo)), insteaad of lineNo
  const canvas = document.querySelector('canvas');
  if (!canvas) return;
  const ctx = canvas.getContext('2d');
  ctx.font = this.font;
  let txt = this.lines.at(lineNo).string;
  let xVals = world.List.new(0.0);
  //let s = 'x-values for line ' + lineNo + ': ';
  for (let i = 0; i < txt.length; i++) {
    let metrics = ctx.measureText(txt.slice(0, i + 1));
    //s += ', ' + metrics.width.toFixed(1);
    xVals.push(metrics.width);
  }
  //console.log(s);
  return xVals;
};
world.TextBox.proto.charSpecForPt = function (worldPt) {
  //if (!this.includesPt(p)) return world.TextCharSpec.new(0, 0, 0, 0);
  let p = worldPt.subPt(this.topLeft);
  let lineIndex = Math.floor((p.y - this.hang) / this.lineHeight);
  let lineY = this.topLeft.y + this.hang + lineIndex * this.lineHeight;
  let xVals = this.xValuesForLine(lineIndex);
  let minDist = 999;
  let xBest = 0;
  let iBest = 0;
  let i = 0;
  xVals.forEach((xVal) => {
    // Find closest x-value
    if (Math.abs(xVal - p.x) < minDist) {
      //Find nearest character x
      xBest = xVal;
      iBest = i;
      minDist = Math.abs(xVal - p.x);
    }
    i++;
  });
  return world.TextCharSpec.new(lineIndex, lineY, xBest + 1, this.lineStarts.at(lineIndex) + iBest);
};
world.TextBox.proto.charSpecForIndex = function (ix) {
  let charSpec = null;
  let lineY = this.topLeft.y + this.hang;
  let lineNo = 0;
  let lineStartslength = this.lineStarts.length();
  // console.log('charSpecForIndex ' + ix + '; lineStartslength = ' + lineStartslength);
  this.lineStarts.forEach((lineStart) => {
    if (
      lineNo == lineStartslength - 1 ||
      (ix >= lineStart && ix < this.lineStarts.at(Math.min(lineNo + 1, lineStartslength - 1)))
    ) {
      let xVals = this.xValuesForLine(lineNo);
      let xBest = xVals.at(ix - lineStart);
      charSpec = world.TextCharSpec.new(lineNo, lineY - 2, xBest + 1, ix);
    }
    lineNo += 1;
    lineY += this.lineHeight;
  });
  if (charSpec == null) return world.TextCharSpec.new(0, 0, 0, 0);
  return charSpec;
};

world.TextBox.proto.acceptKeyboardInput = function (evt) {
  // ** Note selStart and stop may be reversed!
  if (world.copyPasteBuffer == null) world.keybardBuffer = '';
  if (evt.key == 'Shift') return; // redundant keyDown names
  if (evt.key == 'Control') return;
  if (evt.key == 'Alt') return;
  if (evt.key == 'Enter') {
    this.paste(String.fromCharCode(evt.keyCode));
    return;
  }
  if (evt.metaKey || evt.ctrlKey || evt.addLastKey) return this.handleKeyboardShortcuts(evt);

  if (evt.key == 'Backspace') {
    // backspace
    if (this.selectedTextString().length > 0) this.paste(''); // first BS deletes
    else {
      if (this.selStart.strIx >= 1) {
        // subsequent BS deletes backwards
        this.setSelectionRange([this.selStart.strIx - 1, this.selStart.strIx - 1]);
        this.paste('');
      }
    }
    return;
  }
  console.log('calling paste on ' + evt.key);
  this.paste(evt.key); // plain old typing
};
world.TextBox.proto.handleKeyboardShortcuts = function (evt) {
  console.log(evt);
  if ('xcvazdps'.indexOf(evt.key) < 0) return; // so as to not prevent default for some useful cases??
  if (evt.key == 'a') this.setSelectionRange([0, this.string.length - 1]); // SELECT ALL
  if (evt.key == 'c') return (world.copyPasteBuffer = this.selectedTextString()); // COPY
  if (evt.key == 'x') {
    world.copyPasteBuffer = this.selectedTextString(); // CUT
    this.paste('');
  }
  if (evt.key == 'v') {
    if (world.copyPasteBuffer == null) this.paste('??'); // PASTE
    else this.paste(world.copyPasteBuffer);
  }
  if (evt.key == 'z') this.undoReplacement; // UNDO
  if (evt.key == 'd') this.wsEval.call(this.workspaceObj, this.selectedTextString()); // DO IT
  if (evt.key == 's') this.wsEval.call(this.workspaceObj, this.string); // SAVE (eval all)
  if (evt.key == 'p') {
    this.string =
      this.string.slice(0, this.selStop.strIx) +
      this.wsEval.call(this.workspaceObj, this.selectedTextString()).toString() + // PRINT IT
      this.string.slice(this.selStop.strIx);
    this.compose();
  }
  evt.preventDefault();
};
// About replacement...
// There are two states: duringTyping = false or true
// It gets set to false by clicks. It becomes, and remains true after any keystroke
// We save enough state to (a) undo a simple type-over or cmd-V paste,
// ... or (b) repeat the next possible similar repcalement.
world.TextBox.proto.clearTyping = function () {
  this.duringTyping = false;
};
world.TextBox.proto.noteReplacement = function (str) {
  // Copied logic of paste() to save state
  console.log('noteReplacement');
  if (this.duringTyping) {
    this.stringPutIn += str;
    return;
  }
  this.selStartStrix = this.selStart.strIx;
  this.selStopStrix = this.selStop.strIx;
  this.stringTakenOut = this.string.slice(this.selStartStrix, this.selStopStrix);
  this.stringPutIn = str.slice(0);
  this.duringTyping = true;
};
world.TextBox.proto.undoReplacement = function () {
  console.log('undoReplacement');
  // Recreate selection range
  this.selStart = this.charSpecForIndex(this.selStop.strIx - this.stringPutIn.length);
  this.paste(this.stringTakenOut);
  this.selStart = this.charSpecForIndex(this.selStop.strIx - this.stringTakenOut.length);
  this.clearTyping();
};
world.TextBox.proto.redoReplacement = function (str) {
  console.log('redoReplacement');
  // Copied logic of paste() to save state
  this.setStartStrix = this.selStart.strIx;
  this.setStopStrix = this.selStop.strIx;
  this.stringTakenOut = this.string.slice(this.selStart.strIx, this.selStop.strIx);
  this.stringPutIn = str.slice(0);
};
world.TextBox.proto.paste = function (str) {
  // Paste the string over the current selection
  if (!this.duringTyping) this.noteReplacement(str); // copied logic to save state
  let before = this.string.slice(0, this.selStart.strIx);
  let after = this.string.slice(this.selStop.strIx, this.string.length);
  this.string = before + str + after;
  this.compose();
  this.selStart = this.selStop = this.charSpecForIndex(this.selStart.strIx + str.length);
};

world.TextBox.proto.render = function (ctx) {
  ctx.save();
  ctx.fillStyle = this.boxColor.fillStyle;
  ctx.fill();
  ctx.rect(this.topLeft.x, this.topLeft.y, this.extent.x, this.extent.y);
  if (this.borderWidth > 0) {
    ctx.fillStyle = this.borderColor.fillStyle;
    ctx.stroke();
  }
  ctx.clip(); // clip to this rectangle
  // -- turn dropshadows off for the text part of this render
  ctx.shadowColor = 'transparent'; //
  ctx.shadowBlur = 0; //
  ctx.shadowOffsetX = 0; //
  ctx.shadowOffsetY = 0; //
  ctx.fillStyle = this.textColor.fillStyle;
  ctx.font = this.font;
  ctx.textBaseline = 'hanging';
  let lineY = this.topLeft.y + this.hang;
  let nLines = this.lines.length();
  for (let lineNo = 0; lineNo < nLines; lineNo++) {
    ctx.fillStyle = this.selectionColor.fillStyle;
    // *** move this all to a helper function renderSelection()
    let selY = lineY - 2; // selection looks better a bit higher
    if (lineNo == this.selectedLineIndex - 1) {
      // line selection for lists and menus
      ctx.fillRect(this.topLeft.x, selY, this.extent.x, this.lineHeight);
    }
    if (this.selStart != null) {
      // character selection for editing - egad...
      let spec1 = this.selStart;
      let spec2 = this.selStop;
      // Flip stop and start if selection was drawn backwards
      if (
        spec1.lineNo > spec2.lineNo ||
        (spec1.lineNo == spec2.lineNo && spec1.charX > spec2.charX)
      ) {
        spec1 = this.selStop;
        spec2 = this.selStart;
      }
      if (spec1.charX < 1) spec1.charX = 1; // Selection was overwriting border??
      if (spec2.charX < 1) spec2.charX = 1;
      if (lineNo == spec1.lineNo) {
        if (lineNo == spec2.lineNo) {
          // starts and ends on this line
          let hairline = spec2.charX - spec1.charX == 0 ? 2 : 0; // show null selection
          ctx.fillRect(
            spec1.charX + this.topLeft.x,
            selY,
            spec2.charX - spec1.charX + hairline,
            this.lineHeight
          );
        } else {
          // starts here but continues
          ctx.fillRect(
            spec1.charX + this.topLeft.x,
            selY,
            this.extent.x - spec1.charX,
            this.lineHeight
          );
        }
      } else {
        if (lineNo > spec1.lineNo) {
          if (lineNo == spec2.lineNo) {
            // starts above and ends on this line
            ctx.fillRect(this.topLeft.x, selY, spec2.charX, this.lineHeight);
          } else {
            if (lineNo < spec2.lineNo) {
              // starts above and continues
              ctx.fillRect(this.topLeft.x, selY, this.extent.x, this.lineHeight);
            }
          }
        }
      }
    }
    ctx.fillStyle = this.textColor.fillStyle;
    ctx.fillText(this.lines.at(lineNo).string, this.topLeft.x + 2, lineY, 500);
    lineY += this.lineHeight;
  }
  ctx.restore();
};

world.TextBox.proto.selectWord = function (str, i1) {
  // Selection caret before char i1
  if (!str) return i1;
  // Dan's super bracket matching feature
  if (i1 == 0 || i1 == str.length) return [0, str.length - 1]; // select entire string
  var rightBrackets = '*)}]>\'"';
  var leftBrackets = '*({[<\'"';
  function isWhiteSpace(c) {
    return c === '\t' || c === ' ';
  }
  function isAlpha(s) {
    var regEx = /^[a-zA-Z0-9\-]+$/;
    return (s || '').match(regEx);
  }
  function periodWithDigit(c, prev) {
    // return true iff c is a period and prev is a digit
    if (c != '.') return false;
    return '0123456789'.indexOf(prev) >= 0;
  }
  function matchBrackets(str, chin, chout, start, dir) {
    var i = start;
    var depth = 1;
    while (dir < 0 ? i - 1 >= 0 : i + 1 < str.length) {
      i += dir;
      if (str[i] == chin && chin != chout) depth++;
      if (str[i] == chout) depth--;
      if (depth == 0) return i;
    }
    return i;
  }
  function findLine(str, start, dir, endChar) {
    // start points to a CR or LF (== endChar)
    var i = start;
    while (dir < 0 ? i - 1 >= 0 : i + 1 < str.length) {
      i += dir;
      if (str[i] == endChar) return dir > 0 ? [start, i] : [i + 1, start];
    }
    return dir > 0 ? [start + 1, str.length - 1] : [0, start];
  }
  var i2 = i1 - 1;
  if (i1 > 0) {
    // look left for open backets
    if (str[i1 - 1] == '\n' || str[i1 - 1] == '\r') {
      return findLine(str, i1, -1, str[i1]);
    }
    var i = leftBrackets.indexOf(str[i1 - 1]);
    if (str[i1 - 1] == '*' && (i1 - 2 < 0 || str[i1 - 2] != '/')) i = -1; // spl check for /*
    if (i >= 0) {
      var i2 = matchBrackets(str, leftBrackets[i], rightBrackets[i], i1 - 1, 1);
      return [i1, i2 - 1];
    }
  }
  if (i1 < str.length) {
    // look right for close brackets
    if (str[i1] == '\n' || str[i1] == '\r') return findLine(str, i1, -1, str[i1]);
    var i = rightBrackets.indexOf(str[i1]);
    if (str[i1] == '*' && (i1 + 1 >= str.length || str[i1 + 1] != '/')) i = -1; // spl check for */
    if (i >= 0) {
      i1 = matchBrackets(str, rightBrackets[i], leftBrackets[i], i1, -1);
      return [i1 + 1, i2];
    }
  }
  // is a '//' left of me?
  if (str[i1 - 1] === '/' && str[i1 - 2] === '/') {
    while (i2 + 1 < str.length && str[i2 + 1] !== '\n' && str[i2 + 1] !== '\r') {
      i2++;
    }
    return [i1, i2];
  }
  // inside of whitespaces?
  var myI1 = i1;
  var myI2 = i2;
  while (myI1 - 1 >= 0 && isWhiteSpace(str[myI1 - 1])) myI1--;
  while (myI2 < str.length && isWhiteSpace(str[myI2 + 1])) myI2++;
  if (myI2 - myI1 >= 1) return [myI1, myI2];

  var prev = i1 < str.length ? str[i1] : '';
  while (i1 - 1 >= 0 && (isAlpha(str[i1 - 1]) || periodWithDigit(str[i1 - 1], prev))) {
    prev = str[i1 - 1];
    i1--;
  }
  while (i2 + 1 < str.length && (isAlpha(str[i2 + 1]) || periodWithDigit(str[i2 + 1], prev))) {
    prev = str[i2 + 1];
    i2++;
  }
  return [i1, i2];
};

// Theory of copying morphs:
// A simple morph is copied by a copy of all its instance variables.
// But morphs in general comprise a tree of their submorphs
// So a morphCopy is a simpleCopy with its submorph list being a *map*
// of morphs to morphCopies
world.Morph.proto.morphCopy = function () {
  let copy = world.Morph.new(this.fullBounds, this.shape.copy());
  copy.owner = this.owner;
  copy.transform = this.transform.copy(); // may not need to copy
  copy.steppingSpec = this.steppingSpec ? this.steppingSpec.copy() : null;
  copy.submorphs = this.submorphs.map((m) => m.morphCopy());
  return copy;
};

world.WorldMorph.proto.handleStepList = function () {
  // Note this may fire several morphs, not in order if all are due
  // Also (nice) if an error occurs, the morph will be out of stepList
  this.stepList.forEach((spec) => {
    if (spec.nextStepTime < Date.now()) {
      this.stepList = this.stepList.select((s) => s._id != spec._id);
      spec.nextStepTime = Date.now() + spec.stepPeriod;
      if (spec.arg) spec.stepMorph[spec.methodName](spec.arg);
      else spec.stepMorph[spec.methodName]();
      this.stepList.push(spec);
    }
  });
};

world.HandMorph.proto.onPointerDown = function (p, evt) {
  this.hitPoint = p;
  if (evt.shiftKey) {
    // shift drag means copy
    if (this.hasSubmorphs()) return false; // no sense to copy if laden
    else this.addMorph(this.world().topMorphAt(p).morphCopy());
  } else {
    if (this.hasSubmorphs()) this.dropMorph();
    else this.grabMorph();
  }
  return true; // not needed for hands

  // debugger;
  let d = p.subPt(this.location()); // position instead of topLeft?  ***

  this.forEverySubmorph((m) => m.moveBy(d));

  if (this.hasSubmorphs() && this.hitPoint.dist(p) > 2) this.dropMorph();
};
world.HandMorph.proto.grabMorph = function (p, evt) {
  let morphUnder = this.world().topMorphAt(this.location());
  if (morphUnder) this.addMorph(morphUnder); // will remove it from world
};
world.HandMorph.proto.dropMorph = function (p, evt) {
  let morphUnder = this.world().topMorphAt(this.location());
  this.submorphs.forEach((morphToDrop) => {
    if (morphUnder) morphUnder.addMorph(morphToDrop);
    else this.world().addMorph(morphToDrop);
  });
};

world.inspectString = function (obj) {
  let typeStr = typeof obj;
  if (typeStr == 'number') return typeStr + ': ' + obj.toString();
  if (typeStr == 'boolean') return typeStr + ': ' + obj.toString();
  if (typeStr == 'string') return typeStr + ': ' + obj.toString();
  try {
    typeStr = typeStr + ': a' + obj.className;
  } catch (err) {
    typeStr = typeStr + ': ' + err;
  }
  return typeStr;
};

world.SliderMorph = world.Morph.subclass('SliderMorph');
world.SliderMorph.proto.initialize = function (initialBounds, valueFn) {
  world.Morph.proto.initialize.call(this, initialBounds);
  this.value = 0.5; // By convention, my value ranges from 0.0 to 1.0
  this.valExtent = 0.1; // for when showing a range
  this.slider = this.addMorph(world.Morph.new(world.rect(0, 0, 10, 10)));
  this.styleSlider();
  this.slider.onPointerDown = function () {
    return false;
  }; // ignore blipper
  this.emitValueFunction = valueFn;
  console.log(`slider value = ${this.value}, slider = ${this.slider.asString()},
    function = ${this.emitValueFunction}`);
  this.adjustForNewBounds();
};
world.SliderMorph.proto.styleSlider = function () {
  this.slider.setColor(world.Color.red);
};
world.SliderMorph.proto.isVertical = function () {
  var bnds = this.shape.getBounds();
  return bnds.height() > bnds.width();
};
world.SliderMorph.proto.setValueTarget = function (target, msgName) {
  this.valueTarget = target;
  this.valueMessage = msgName;
};
world.SliderMorph.proto.adjustForNewBounds = function () {
  // This method adjusts the slider for changes in value as well as geometry
  var bnds = this.shape.getBounds();
  var ext = this.valExtent;
  if (this.isVertical()) {
    // more vertical...
    var elevPix = Math.max(ext * bnds.height(), 6); // height of elevator in pixels
    var topLeft = world.pt(0, (bnds.height() - elevPix) * this.value);
    var sliderExt = world.pt(bnds.width(), elevPix);
  } else {
    // more horizontal...
    var elevPix = Math.max(ext * bnds.width(), 6); // thickness of elevator in pixels
    var topLeft = world.pt((bnds.width() - elevPix) * this.value, 0);
    var sliderExt = world.pt(elevPix, bnds.height());
  }
  console.log('isVertical =  ' + this.isVertical());
  var bb = bnds.topLeft.addPt(topLeft).extent(sliderExt);
  console.log('adjust bounds ' + bb.asString());
  this.slider.setBounds(bnds.topLeft.addPt(topLeft).extent(sliderExt));
  console.log('adjust done');
};
world.SliderMorph.proto.onPointerDown = function (pt, evt) {
  //	Note: want setMouseFocus to also cache the transform and record the hitPoint.
  //	Ideally thereafter only have to say, eg, morph moveTo: evt.hand.adjustedMousePoint
  if (this.actorId != null && this.actorId != window.actorId) return false;
  if (!this.includesPt(pt)) return false;
  this.hitPoint = pt;
  this.world().setPointerFocus(this);
  return true;
};
world.SliderMorph.proto.onPointerUp = function (pt, evt) {
  if (this.actorId != null && this.actorId != window.actorId) return false;
  if (!this.includesPt(pt)) return false;
  this.hitPoint = null;
  this.world().setPointerFocus(null);
  return true;
};
world.SliderMorph.proto.onPointerMove = function (pt, evt) {
  if (this.actorId != null && this.actorId != window.actorId) return false;
  if (!this.hitPoint) return false;
  if (!this.includesPt(pt)) return false;
  // Compute a new value from a new mouse point, and emit it
  console.log('----pointerMove-----');
  var bnds = this.shape.getBounds();
  let ext = this.valExtent; // for now...
  let p = pt.subPt(bnds.topLeft);
  if (this.isVertical()) {
    // more vertical...
    var elevPix = Math.max(ext * bnds.height(), 6); // thickness of elevator in pixels
    var newValue = (p.y - elevPix / 2) / (bnds.height() - elevPix);
  } else {
    // more horizontal...
    var elevPix = Math.max(ext * bnds.width(), 6); // thickness of elevator in pixels
    var newValue = (p.x - elevPix / 2) / (bnds.width() - elevPix);
  }
  console.log('this.emitValueFunction.call(this, newValue' + newValue);
  //debugger;
  this.value = this.clipValue(newValue);
  // *** Note that emitValueFunction is not binding the right target
  if (this.valueTarget) this.valueTarget[this.valueMessage](this.value);
  else this.emitValueFunction.call(this.owner, this.value);
  console.log("moving slider's owner is " + this.owner.asString());
  if (this.owner.className == 'ListPane') {
    console.log('... whose list = ' + this.owner.contentPane.itemList.first() + '...');
  }
  this.adjustForNewBounds();
  return true;
};
world.SliderMorph.proto.handlesMouseDown = function (evt) {
  return true;
};
world.SliderMorph.proto.clipValue = function (val) {
  return Math.min(1.0, Math.max(0, 0, val)).toFixed(3);
};

world.ScrollPane.proto.initialize = function (initialBounds) {
  world.Morph.proto.initialize.call(this, initialBounds);
  this.contentPane = null; // filled in by subclasses
  this.scrollBar = null; // filled in by subclasses
};
world.ScrollPane.proto.renderOn = function (ctx) {
  let bnds = this.getBounds();
  ctx.save();
  ctx.rect(bnds.topLeft.x, bnds.topLeft.y, bnds.extent.x, bnds.extent.y);
  ctx.clip(); // clip to this pane rectangle
  world.Morph.proto.renderOn.call(this, ctx);
  ctx.restore();
};
world.ScrollPane.proto.getScrollPosition = function () {
  var ht = this.contentPane.getBounds().height;
  var slideRoom = ht - this.getBounds().height;
  return -this.contentPane.transform.translation.y / slideRoom;
};
world.ScrollPane.proto.setScrollPosition = function (scrollPos) {
  var ht = this.contentPane.getBounds().height();
  var slideRoom = ht - this.getBounds().height();
  console.log(this.contentPane.asString() + 'ScrollPane.proto.setScrollPosition = ' + scrollPos);
  this.contentPane.transform.translation.y = Math.min(0, -slideRoom * scrollPos);
};
world.ScrollPane.proto.getVisibleExtent = function (scrollPos) {
  return Math.min(1, this.getBounds().height / Math.max(10, this.contentPane.getBounds().height));
};
world.ScrollPane.proto.scrollToTop = function () {
  this.setScrollPosition(0);
  this.scrollBar.adjustForNewBounds();
};

  //world.initLively(); // build a simple world with a single rectangle morph

world.HaloMorph.proto.pointerDownOnHandle = function (handle, pt, evt) {
  switch (handle.handleName) {
    case 'Rotate':
      this.target.rotateFrom(handle.hitPoint);
      break;
    case 'Style':
      this.target().restyle();
      break;
    case 'Copy':
      let m = this.world().addMorph(this.target.morphCopy());
      m.moveBy(world.pt(20, -20));
      m.showHalo();
      break;
    case 'Menu':
      this.world().showWorldMenuAt(handle.getBounds().center());
      break;
    case 'Grab':
      this.target.owner.submorphs = this.target.owner.submorphs.promote(this.target);
      this.target.dragFrom(pt, evt);
      break;
    case 'Drag':
      this.target.dragFrom(pt, evt);
      break;
    case 'Delete':
      this.target.remove();
      break;
    case 'Browse':
      let browser = world.PanelMorph.test();
      browser.classPane.setSelectionString(this.target.className);
      break;
    case 'Inspect':
      this.target.inspect();
      break;
    case 'Scale':
      this.target.resizeFrom(handle.hitPoint);
      break;
    default:
      console.log('Invalid halo handle name??');
  }
  this.remove();
};
world.Morph.proto.showHalo = function () {
  this.world().removeExistingHalos();
  // console.log('showHalo() on ' + this.asString());
  this.world().addMorph(world.HaloMorph.new(this));
};

};
world.WorldMorph.proto.removeExistingHalos = function () {
  this.submorphs.forEach((morph) => {
    if (morph.className == 'HaloMorph') morph.remove();
  });
};

world.HaloHandle = world.Morph.subclass('HaloHandle');
world.HaloHandle.proto.initialize = function (index, iconLetter, handleName, halo) {
  world.Morph.proto.initialize.call(this, null, this.makeHandleShape(index, iconLetter, halo));
  this.halo = halo;
  this.handleName = handleName;
  let letterMorph = this.addMorph(
    world.TextMorph.new(this.getBounds().movedBy(world.pt(3, 0)), iconLetter)
  );
  letterMorph.shape.setStyles(null, 0, null);
};
world.HaloHandle.proto.makeHandleShape = function (index, iconLetter, halo) {
  let w = 20; // handle width
  let frame = halo.getBounds().insetBy(0 - w);
  let p1 = frame.bottomLeft().subPt(world.pt(0, w));
  let p2 = frame.topLeft;
  let p3 = frame.topRight().subPt(world.pt(w, 0));
  let p4 = frame.bottomRight().subPt(world.pt(w, w));
  let c1,
    c2,
    d = null;
  if (index <= 4) {
    c1 = p1;
    c2 = p2;
    d = (4 - index) / 3;
  } else if (index <= 7) {
    c1 = p2;
    c2 = p3;
    d = (7 - index) / 3;
  } else {
    c1 = p3;
    c2 = p4;
    d = (10 - index) / 3;
  }
  let handleLoc = c2.addPt(c1.subPt(c2).scaleBy(d));
  let radius = w / 2;
  let handleShape = world.Ellipse.new(handleLoc.addPt(world.pt(radius, radius)), radius);
  handleShape.setStyles(null, 1, world.Color.blue);
  return handleShape;
};
world.HaloHandle.proto.onPointerDown = function (pt, evt) {
  this.halo.pointerDownOnHandle(this, pt, evt);
};
